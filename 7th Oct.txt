import java.util.Scanner;

public class Switch_demo {

	public static void main(String[] args) 
	{
		Scanner sc=new Scanner(System.in);
		int choice;
		
		System.out.println("1:New \"Pin\"\t generate\n2:Withdraw \n3:Blance Check\n4:Pin change");
		choice=sc.nextInt();
		
		switch(choice)
		{
		case 1:
			System.out.println("The process of new pin generate");
			break;
		case 2:
			System.out.println("Welcome to My bank \n The process of money withdraw");
			break;
		case 3:
			System.out.println("The process of balance Check");
			break;
		case 4:
			System.out.println("The process of pin change");
			break;
		default:
			System.out.println("Invvalid input");
			break;
		}

	}

}

import java.util.Scanner;

public class Nested_demo {

		age=sc.nextInt();
		System.out.println("Enter your gender");
		gender=sc.next().charAt(0);
		
		if(age>=50)
		{
			if(gender=='F'||gender=='f')

				System.out.println("9% ROI");
			}

		}
